SELECT I.REST_ID, I.REST_NAME, I.FOOD_TYPE, I.FAVORITES, I.ADDRESS, ROUND(SUM(R.REVIEW_SCORE)/COUNT(I.REST_ID), 2) AS SCORE
FROM REST_INFO I
INNER JOIN REST_REVIEW R ON R.REST_ID = I.REST_ID
WHERE I.ADDRESS LIKE '서울특별시%' OR I.ADDRESS LIKE '서울시%'
GROUP BY I.REST_ID
ORDER BY SCORE DESC, FAVORITES DESC
