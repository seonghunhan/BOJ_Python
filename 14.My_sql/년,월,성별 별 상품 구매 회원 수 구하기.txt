SELECT YEAR(B.SALES_DATE) AS YEAR, MONTH(B.SALES_DATE) AS MONTH, A.GENDER, COUNT(DISTINCT B.USER_ID) AS USERS
FROM ONLINE_SALE B
JOIN USER_INFO A ON A.USER_ID = B.USER_ID
WHERE A.GENDER IS NOT NULL
GROUP BY YEAR(B.SALES_DATE), MONTH(B.SALES_DATE), A.GENDER
ORDER BY YEAR(B.SALES_DATE), MONTH(B.SALES_DATE), A.GENDER

SELECT DATE_FORMAT(B.SALES_DATE, '%Y') AS YEAR, DATE_FORMAT(B.SALES_DATE, '%m'), A.GENDER, COUNT(DISTINCT A.USER_ID)AS USER
FROM USER_INFO A
INNER JOIN ONLINE_SALE B ON B.USER_ID = A.USER_ID
WHERE A.GENDER IS NOT NULL
GROUP BY DATE_FORMAT(B.SALES_DATE, '%Y'), DATE_FORMAT(B.SALES_DATE, '%m'), A.GENDER
ORDER BY DATE_FORMAT(B.SALES_DATE, '%Y'), DATE_FORMAT(B.SALES_DATE, '%m'), A.GENDER
